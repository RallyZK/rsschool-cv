Code example

Task: to write a function which formats a duration, given as a number of seconds, in a human-friendly way.

function formatDuration (seconds) {
let y, d, h, m, s, res = '';
y = Math.floor(seconds / (31536000));
d = Math.floor((seconds - y * 31536000) / (86400));
h = Math.floor((seconds - y * 31536000 - d * 86400) / (3600));
m = Math.floor((seconds - y * 31536000 - d * 86400 - h * 3600) / (60));
s = seconds - y * 31536000 - d * 86400 - h * 3600 - m * 60;
let arr = [[y, 'year'], [d, 'day'], [h, 'hour'], [m, 'minute'], [s, 'second']]
let resarr = arr.filter(el => el[0] !== 0)
resarr.forEach(el => {
if (el[0] !== 1) {
el[1] = el[1] + 's'
}
})
if (resarr.length === 0) {
res = 'now';
} else {
let nokta = ', '
for (let i = 0; i < resarr.length; i++) {
if (i === resarr.length - 2) {
nokta = ' and '
res = res + resarr[i][0] + ' ' + resarr[i][1] + nokta
} else if (i === resarr.length - 1) {
res = res + resarr[i][0] + ' ' + resarr[i][1]
} else {
res = res + resarr[i][0] + ' ' + resarr[i][1] + nokta
}
}
}
return res
}
